### 문자열의 간단한 연산

	#include<iostream>
	#include<string>
	using namespace std;

	int main()
	{
		//문자열 끼리 더하기
		string a("Hello");
		string b("World");

		cout << a + b << endl;
		stign hw = a + b;
		hw += "I'm good";
		
		cout << hw << endl;		// string에 문자열을 더하는 것을 append라고 한다.

		//문자열 길이 측정
		string a("Hello, world")

		cout << a.length() << endl;

		return 0;
