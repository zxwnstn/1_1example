### struct 변수의 초기화 의미
	struct Person
	{
		double	height = 3.0
		float	weight = 200.0
		int		age	   = 100;
		string name	   = "Mr. Incredible"
	}; // 초기화를 직접한 스트럭트

	int main()
	{
		Person me;
